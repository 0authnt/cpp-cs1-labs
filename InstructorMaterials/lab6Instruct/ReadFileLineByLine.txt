#include <fstream>
#include <string>
#include <iostream>

using std::string;
using std::ifstream; using std::ofstream; 
using std::cin; using std::cout; using std::endl;

int main(){

   // inputing file name
    cout << "Input file name to print: ";
    string fileName;
    cin >> fileName;

    // opening file
    ifstream fin(fileName);

    // printing
    string line;
    while(getline(fin,line))
       cout << line << endl;

    fin.close();
}
